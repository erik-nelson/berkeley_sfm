00049 void AnnotateFeatures(const FeatureList& features, Image& image,
00050                       unsigned int radius, unsigned int line_thickness) {
00051   // Create a random number generator for random colors.
00052   math::RandomGenerator rng(math::RandomGenerator::Seed());
00053 
00054   // Get OpenCV mat from the input image.
00055   cv::Mat cv_image;
00056   image.ToCV(cv_image);
00057 
00058   for (const auto& feature : features) {
00059     cv::Point cv_feature;
00060     cv_feature.x = feature.u_;
00061     cv_feature.y = feature.v_;
00062 
00063     const cv::Scalar color(rng.Double(), rng.Double(), rng.Double());
00064     cv::circle(cv_image, cv_feature, radius, color, line_thickness);
00065   }
00066 
00067   // Store the OpenCV mat in the image.
00068   image.FromCV(cv_image);

00112     feature2.y = feature_match.feature2_.v_;

00114

00119     // Draw circles around each feature.
