00089 bool EightPointAlgorithmSolver::ComputeFundamentalMatrix(
00090     const FeatureMatchList& matched_features,
00091     Eigen::Matrix3d& fundamental_matrix) const {
00092   // Following: https://www8.cs.umu.se/kurser/TDBD19/VT05/reconstruct-4.pdf
00093 
00094   // First make sure we even have enough matches to run the eight-point
00095   // algorithm.
00096   if (matched_features.size() < 8) {
00097     VLOG(1) << "Cannot use the eight-point algorithm with less than 8 feature "
00098                "matches.";
00099     return false;
00100   }
00101 
00102   // Build the A matrix from matched features.
00103   Eigen::MatrixXd A;
00104   A.resize(matched_features.size(), 9);
00105 
00106   // If requested, normalize feature positions prior to computing the A matrix.
00107   Eigen::Matrix3d T1(Eigen::MatrixXd::Identity(3, 3));
00108   Eigen::Matrix3d T2(Eigen::MatrixXd::Identity(3, 3));
00109   if (options_.normalize_features) {
00110     T1 = ComputeNormalization(matched_features, true /*feature set 1*/);
00111     T2 = ComputeNormalization(matched_features, false /*feature set 2*/);
00112   }
00113 
00114   // Incrementally add rows to A.
00115   for (size_t ii = 0; ii < matched_features.size(); ++ii) {
00116     double u1 = matched_features[ii].feature1_.u_;
00117     double v1 = matched_features[ii].feature1_.v_;
00118     double u2 = matched_features[ii].feature2_.u_;
00119     double v2 = matched_features[ii].feature2_.v_;
00120 
00121     if (options_.normalize_features) {
00122       u1 = T1(0, 0) * u1 + T1(0, 2);
00123       v1 = T1(1, 1) * v1 + T1(1, 2);
00124       u2 = T2(0, 0) * u2 + T2(0, 2);
00125       v2 = T2(1, 1) * v2 + T2(1, 2);
00126     }
00127 
00128     A(ii, 0) = u1*u2;
00129     A(ii, 1) = u1*v2;
00130     A(ii, 2) = u1   ;
00131     A(ii, 3) = v1*u2;
00132     A(ii, 4) = v1*v2;
00133     A(ii, 5) = v1   ;
00134     A(ii, 6) = u2   ;
00135     A(ii, 7) = v2   ;
00136     A(ii, 8) = 1    ;
00137   }
00138 
00139   // Get svd(A). Save some time and compute a thin U. We still need a full V.
00140   Eigen::JacobiSVD<Eigen::MatrixXd> svd;
00141   svd.compute(A, Eigen::ComputeThinU | Eigen::ComputeFullV);
00142   if (!svd.computeV()) {
00143     VLOG(1) << "Failed to compute a singular value decomposition of A matrix.";
00144     return false;
00145   }
00146 
00147   // Get the fundamental matrix elements from the SVD decomposition.
00148   const Eigen::VectorXd f_vec = svd.matrixV().col(8);
00149 
00150   // Turn the elements of the fundamental matrix into an actual matrix.
00151   fundamental_matrix.row(0) = f_vec.topRows(3).transpose();
00152   fundamental_matrix.row(1) = f_vec.middleRows(3, 3).transpose();
00153   fundamental_matrix.row(2) = f_vec.bottomRows(3).transpose();
00154 
00155   // If requested, make sure that the computed fundamental matrix has rank 2.
00156   // This is step 2 of the eight-point algorithm from the slides.
00157   if (options_.enforce_fundamental_matrix_rank_deficiency) {
00158     // Get svd(F). We need full U and V to reconstruct the rank deficient F.
00159     svd.compute(fundamental_matrix, Eigen::ComputeFullU | Eigen::ComputeFullV);
00160     svd.compute(fundamental_matrix);
00161     if (!svd.computeU() || !svd.computeV()) {
00162       VLOG(1) << "Failed to compute a singular value decomposition of "
00163                  "fundamental matrix.";
00164       return false;
00165     }
00166 
00167     // Build a matrix of the first 8 singular values down the diagonal. Make the
00168     // last diagonal entry 0.
00169     Eigen::MatrixXd S_deficient(Eigen::MatrixXd::Zero(3, 3));
00170     S_deficient(0, 0) = svd.singularValues()(0);
00171     S_deficient(1, 1) = svd.singularValues()(1);
00172     fundamental_matrix = svd.matrixU() * S_deficient * svd.matrixV().transpose();
00173   }
00174 
00175   // If normalization was requested, we need to 'un-normalize' the fundamental
00176   // matrix.
00177   if (options_.normalize_features) {
00178     fundamental_matrix = T1.transpose() * fundamental_matrix * T2;
00179   }
00180 
00181   return true;
00182 }

00184 Eigen::Matrix3d EightPointAlgorithmSolver::ComputeNormalization(
00185     const FeatureMatchList& matched_features, bool use_feature_set1) const {
00186   // Compute a mean translation from the origin.
00187   double mean_u = 0.0;
00188   double mean_v = 0.0;
00189   for (size_t ii = 0; ii < matched_features.size(); ++ii) {
00190     if (use_feature_set1) {
00191       mean_u += matched_features[ii].feature1_.u_;
00192       mean_v += matched_features[ii].feature1_.v_;
00193     } else {
00194       mean_u += matched_features[ii].feature2_.u_;
00195       mean_v += matched_features[ii].feature2_.v_;
00196     }
00197   }
00198   mean_u /= static_cast<double>(matched_features.size());
00199   mean_v /= static_cast<double>(matched_features.size());
00200 
00201   // Compute a scale factor such that after translation, all points will be an
00202   // average distance of sqrt(2) away from the origin.
00203   // This uses Eqs. 1.28 - 1.37 from here:
00204   // http://www.ecse.rpi.edu/Homepages/qji/CV/8point.pdf
00205   double scale = 0.0;
00206   for (size_t ii = 0; ii < matched_features.size(); ++ii) {
00207     double u = 0.0, v = 0.0;
00208     if (use_feature_set1) {
00209       u = matched_features[ii].feature1_.u_;
00210       v = matched_features[ii].feature1_.v_;
00211     } else {
00212       u = matched_features[ii].feature2_.u_;
00213       v = matched_features[ii].feature2_.v_;
00214     }
00215 
00216     scale += sqrt(((u - mean_u)*(u - mean_u) + (v - mean_v)*(v-mean_v)) / 2.0);
00217   }
00218   scale /= static_cast<double>(matched_features.size());
00219 
00220   // If the calculated scale, for some reason, is ridiculously small (i.e.
00221   // dividing by zero), return an identity output matrix. This will make us not
00222   // normalize the data, but we should still get an okay fundamental matrix.
00223   if (scale < 1e-4) {
00224     LOG(WARNING) << "When normalizing feature coordinates for fundamental "
00225                     "matrix computation, scale factor was nearly zero. "
00226                     "Proceeding without normalizing data.";
00227     return Eigen::MatrixXd::Identity(3, 3);
00228   }
00229 
00230   // Populate the output matrix.
00231   Eigen::Matrix3d T(Eigen::MatrixXd::Identity(3, 3));
00232   T(0, 0) = 1.0 / scale;
00233   T(1, 1) = 1.0 / scale;
00234   T(0, 2) = -mean_u / scale;
00235   T(1, 2) = -mean_v / scale;
00236 
00237   return T;
00238 }
