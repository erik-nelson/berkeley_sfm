00086  protected:
00087   // The matched image data contains a list of matched features for a set of

00091   PairwiseImageMatchList matched_image_data_;
00092 
00093   // A set of options used for computing the fundamental matrix.

00123 bool FundamentalMatrixSolver::ComputeFundamentalMatrices(
00124     std::vector<Eigen::Matrix3d>& fundamental_matrices) {
00125   // Clear the output.
00126   fundamental_matrices.clear();
00127 
00128   // Determine a fundamental matrix for each pair of images.
00129   for (const auto& pair_data : matched_image_data_) {
00130     Eigen::Matrix3d fundamental_matrix;
00131     if (ComputeFundamentalMatrix(pair_data.feature_matches_,
00132                                  fundamental_matrix)) {
00133       fundamental_matrices.push_back(fundamental_matrix);
00134     } else {
00135       VLOG(1) << "Failed to compute funamental matrix between images "
00136               << pair_data.image1_index_ << " and " << pair_data.image2_index_
00137               << ".";
00138     }
00139   }
00140 
00141   // Return whether or not we computed any fundamental matrices.
00142   if (fundamental_matrices.empty()) {
00143     VLOG(1) << "Unable to compute a fundamental matrix for any of the input "
00144                "image pairs.";
00145     return false;
00146   }
00147 
00148   return true;
00149 }
