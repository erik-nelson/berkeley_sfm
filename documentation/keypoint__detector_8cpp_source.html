00048 bool KeypointDetector::SetDetector(const std::string& detector_type) {
00049   // Set the detector type.
00050   detector_type_ = detector_type;
00051   bool valid_detector_type = true;
00052 
00053   // Create an OpenCV feature detector.
00054   if (detector_type.compare("SIFT") == 0) {
00055     detector_ = cv::FeatureDetector::create("SIFT");
00056   } else if (detector_type.compare("SURF") == 0) {
00057     detector_ = cv::FeatureDetector::create("SURF");
00058   } else if (detector_type.compare("FAST") == 0) {
00059     detector_ = cv::FeatureDetector::create("FAST");
00060   } else if (detector_type.compare("STAR") == 0) {
00061     detector_ = cv::FeatureDetector::create("STAR");
00062   } else if (detector_type.compare("ORB") == 0) {
00063     detector_ = cv::FeatureDetector::create("ORB");
00064   } else if (detector_type.compare("BRISK") == 0) {
00065     detector_ = cv::FeatureDetector::create("BRISK");
00066   } else if (detector_type.compare("MSER") == 0) {
00067     detector_ = cv::FeatureDetector::create("MSER");
00068   } else if (detector_type.compare("GFTT") == 0) {
00069     detector_ = cv::FeatureDetector::create("GFTT");
00070   } else if (detector_type.compare("HARRIS") == 0) {
00071     detector_ = cv::FeatureDetector::create("HARRIS");
00072   } else if (detector_type.compare("DENSE") == 0) {
00073     detector_ = cv::FeatureDetector::create("Dense");
00074   } else if (detector_type.compare("SIMPLEBLOB") == 0) {
00075     detector_ = cv::FeatureDetector::create("SimpleBlob");
00076   } else {
00077     VLOG(1) << "Detector type \"" << detector_type
00078             << "\"is not available. Defaulting to FAST.";
00079     detector_ = cv::FeatureDetector::create("FAST");
00080     valid_detector_type = false;
00081     detector_type_ = "FAST";
00082   }
00083 
00084   return valid_detector_type;
00085 }

00087 bool KeypointDetector::DetectKeypoints(const Image& image,
00088                                        KeypointList& keypoints_out) {
00089   // Make the user has called SetDetector().
00090   if (detector_type_.empty()) {
00091     VLOG(1) << "Detector has not been specified via SetDetector(). Failed to "
00092                "detect keypoints.";
00093     return false;
00094   }
00095 
00096   CHECK(detector_) << "The feature detector is null.";
00097 
00098   // Convert the input image to OpenCV's format. Note that features must be
00099   // detected on the grayscale image, and that the image format must be CV_8U.
00100   cv::Mat cv_image;
00101   image.ToCV(cv_image);
00102   cv_image.convertTo(cv_image, CV_8U, 255);
00103 
00104   // Detect keypoints in the image.
00105   try {
00106     detector_->detect(cv_image, keypoints_out);
00107   } catch (const std::exception& e) {
00108     VLOG(1) << "Failed to detect keypoints: " << e.what();
00109     return false;
00110   }
00111 
00112   return true;
00113 }
